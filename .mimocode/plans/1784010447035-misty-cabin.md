## Fix 2: Assignment DTO Validation
**File:** `backend/src/modules/issues/dto/assign-issue.dto.ts`
**Current:** Both fields optional, allows empty requests
**Fix:** Add custom validator requiring at least one field
**Business Logic Preserved:** Same assignment behavior, just rejects empty requests

```typescript
import { IsString, IsOptional, ValidateIf, ValidationOptions, registerDecorator } from 'class-validator';

function AtLeastOneTarget(validationOptions?: ValidationOptions) {
  return function (object: Object, propertyName: string) {
    registerDecorator({
      name: 'atLeastOneTarget',
      target: object.constructor,
      propertyName: propertyName,
      options: validationOptions,
      validator: {
        validate(value: any, args: any) {
          const obj = args.object;
          return !!(obj.targetUserId || obj.targetOrgId);
        },
        defaultMessage(args: any) {
          return 'At least one of targetUserId or targetOrgId must be provided';
        },
      },
    });
  };
}

export class AssignIssueDto {
  @IsString()
  @IsOptional()
  targetUserId?: string;

  @IsString()
  @IsOptional()
  targetOrgId?: string;

  @AtLeastOneTarget({ message: 'At least one of targetUserId or targetOrgId must be provided' })
  _atLeastOne: any; // Virtual field for validation
}
```

**Alternative Simpler Fix:** Add validation in service layer instead:
